    def generate_swap_params(self) -> Tuple[str, bytes, int]:
        """Generate secure parameters for a new atomic swap."""
        # Generate secure random preimage
        preimage = secrets.token_bytes(32)
        # Create hash of preimage for HTLC
        hash_value = hashlib.sha256(preimage).digest()
        # Set timelock values (Mars timelock < BTC timelock to ensure safety)
        # This ensures that the receiver must reveal the secret first
        current_time = int(time.time())
        timelock = current_time + self.config.timelock_duration
        
        # Log parameters with secure logging (no sensitive data)
        self.logger.info(f"Generated new swap parameters: hash={hash_value.hex()}, " 
                          f"timelock={timelock}")
        
        return preimage.hex(), hash_value, timelock
    
    def create_bitcoin_htlc(self, recipient_addr, amount_btc, hash_value, timelock) -> str:
        """Create Bitcoin HTLC transaction."""
        # Validate parameters
        self._validate_address(recipient_addr, "Bitcoin")
        self._validate_amount(amount_btc, "Bitcoin")
        
        # Create HTLC script
        # if hash_preimage == hash_value: allow recipient to claim
        # else if current_time > timelock: allow refund
        htlc_script = self.btc_node.create_htlc_script(
            recipient_address=recipient_addr,
            sender_address=self.config.btc_refund_address,
            hash_value=hash_value,
            timelock=timelock
        )
        
        # Fund transaction from hot wallet
        tx_id = self.btc_node.create_and_broadcast_transaction(
            output_script=htlc_script,
            amount=amount_btc,
            fee_rate=self.config.btc_fee_rate
        )
        
        self.logger.info(f"Created Bitcoin HTLC: tx_id={tx_id}, recipient={recipient_addr}")
        return tx_id
    
    def create_marscoin_htlc(self, recipient_addr, amount_mars, hash_value, timelock) -> str:
        """Create Marscoin HTLC transaction with same hash but earlier timelock."""
        # Mars timelock is earlier to ensure recipient must reveal the secret first on Mars
        # before the sender can claim a refund on Bitcoin
        mars_timelock = timelock - self.config.timelock_safety_margin
        
        # Similar pattern to Bitcoin HTLC creation
        htlc_script = self.mars_node.create_htlc_script(
            recipient_address=recipient_addr,
            sender_address=self.config.mars_refund_address,
            hash_value=hash_value,
            timelock=mars_timelock
        )
        
        tx_id = self.mars_node.create_and_broadcast_transaction(
            output_script=htlc_script,
            amount=amount_mars,
            fee_rate=self.config.mars_fee_rate
        )
        
        self.logger.info(f"Created Marscoin HTLC: tx_id={tx_id}, recipient={recipient_addr}")
        return tx_id
    
    def monitor_swap(self, btc_tx_id, mars_tx_id, hash_value, timelock) -> bool:
        """Monitor and finalize the atomic swap."""
        # Set up monitoring for both chains

2. P2P Communication Layer
pythonclass P2PSwapProtocol:
    """Encrypted P2P communication for swap execution"""
    
    def __init__(self, peer_pubkey):
        self.peer_pubkey = peer_pubkey
        self.session_key = None
        
    def establish_connection(self):
        """ECDH key exchange for encrypted channel"""
        # Generate ephemeral keypair
        my_ephemeral = generate_keypair()
        
        # Exchange pubkeys
        peer_ephemeral = self.exchange_pubkeys(my_ephemeral.pubkey)
        
        # Derive shared secret
        self.session_key = ecdh(my_ephemeral.privkey, peer_ephemeral)
        
        return encrypted_channel(self.session_key)
    
    def swap_handshake(self, offer):
        """Negotiate swap parameters"""
        # Send acceptance
        self.send({
            'type': 'accept_offer',
            'offer_id': offer['id'],
            'btc_address': self.get_btc_address()
        })
        
        # Receive confirmation
        response = self.receive()
        
        if response['type'] == 'confirmed':
            return self.execute_swap(offer, response)
        else:
            raise SwapRejected(response['reason'])

            mars_htlc = self.wait_for_mars_htlc()
            
            # 2. Extract hash from MARS HTLC
            hash = self.extract_hash(mars_htlc)
            
            # 3. Create BTC HTLC with same hash
            btc_htlc = self.create_btc_htlc(hash)
            self.broadcast(btc_htlc)
            
            # 4. Monitor BTC chain for secret reveal
            secret = self.monitor_btc_for_secret()
            
            # 5. Claim MARS using secret
            self.claim_mars(secret)

            'timestamp': time.time(),

                sign(swap_id, participant_a),
                sign(swap_id, participant_b)
            ]

        # Publish to DHT
        for participant in participants:
            self.dht.append(
                f"reputation/{participant}", 
                attestation
            )
